// Cyclomatic complexity 3
class A {
    B b = new B();

    int doMagicA(int value1, int value2, int value3) {
        if (value1 == 0) {
            return 1 + b.doMagicB(value2, value3);
        } else if (value1 > 0) {
            return value1 + 2 * b.doMagicB(value2, value3);
        } else {
            return value1 - 2 * b.doMagicB(value2, value3);
        }
    }
}
